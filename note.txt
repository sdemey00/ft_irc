#pragma once
#include <string>

class IConnection;

class User {
public:
    explicit User(IConnection* conn);

    void setNick(const std::string&);
    const std::string& getNick() const;

    bool isRegistered() const;
    void setRegistered(bool);

    void send(const std::string& msg);

private:
    std::string _nick;
    bool _registered;
    IConnection* _conn;
};
User::User(IConnection* conn)
    : _registered(false), _conn(conn) {}

void User::send(const std::string& msg) {
    if (_conn)
        _conn->send(msg);
}
